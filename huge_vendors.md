SQL used:

```
WITH vendor_packages AS (
    SELECT
        split_part(s.package, '/', 1) AS vendor,
        s.disk_space / 1024.0 AS size_mb,
        COALESCE(p.github_stars, 0) AS stars
    FROM package_stats s
    JOIN packagist_stats p ON p.package = s.package
    WHERE s.package LIKE '%/%'
),
vendor_stats AS (
    SELECT
        vendor,
        count(*) AS repo_count,
        sum(size_mb) AS total_mb,
        sum(stars) AS total_stars
    FROM vendor_packages
    GROUP BY vendor
)
SELECT
    vendor,
    repo_count,
    round(total_mb::numeric, 2) AS total_mb,
    total_stars,
    round(
        (
            total_mb / NULLIF(total_stars + 1, 0)
        )::numeric,
        4
    ) AS badness_score
FROM vendor_stats
WHERE total_mb >= 250
ORDER BY
    badness_score DESC,
    total_mb DESC,
    repo_count ASC;
```
